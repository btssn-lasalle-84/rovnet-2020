182  QString trameTelemetrie, trameCapteur;
183 
184  QStringList trames = trame.split("\r\n");
185  qDebug() << Q_FUNC_INFO << "trame" << trames.size();
186  for(int i = 0; i < trames.size(); i++)
187  {
188  if(!trames[i].isEmpty())
189  {
190  if(trames[i].startsWith(DEBUT_TRAME_TELEMETRIE))
191  {
192  trameTelemetrie = trames[i];
193  decoderTrameTelemetrie(trameTelemetrie);
194  }
195  else if(trames[i].startsWith(DEBUT_TRAME_CAPTEUR))
196  {
197  trameCapteur = trames[i];
198  decoderTrameCapteur(trameCapteur);
199  }
200  }
201  else

203  qDebug() << Q_FUNC_INFO << "Trame inconnue !";
204  }
