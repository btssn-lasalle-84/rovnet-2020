132 void IHMReglageVideo::chargerListeCameraDisponible()
133 {
134  int nbCameras = QCameraInfo::availableCameras().count();
135  qDebug() << Q_FUNC_INFO << "Caméra(s) disponible(s)" << QCameraInfo::availableCameras().count();
136  if (nbCameras > 0)
137  {
138  #ifndef SANS_DETECTION
139  QList<QCameraInfo> cameras = QCameraInfo::availableCameras();
140  listeCameras->clear();
141  int choix = -1, i = 0;
142  foreach (const QCameraInfo &cameraInfo, cameras)
143  {
144  listeCameras->addItem(cameraInfo.deviceName());
145  qDebug() << Q_FUNC_INFO << "Device" << cameraInfo.deviceName();
146  qDebug() << Q_FUNC_INFO << "Description" << cameraInfo.description();
147  if(rov->getCamera() != nullptr)
148  {
149  if(cameraInfo.deviceName() == rov->getCamera()->getNom())
150  {
151  choix = i;
152  }
153  }
154  i++;
155  }
156  #else
157  int choix = 0;
158  qDebug() << Q_FUNC_INFO << "Device" << rov->getCamera()->getNom();
159  listeCameras->addItem(rov->getCamera()->getNom());
160  #endif
161 
162  if(choix != -1)
163  {
164  if(rov->getCamera() != nullptr)
165  chargerListeResolutionDisponible(rov->getCamera()->getNom());
166  listeCameras->setCurrentIndex(choix);
167  }
168  }
169 
170 }

179 void IHMReglageVideo::chargerListeResolutionDisponible(QString nom)
180 {
