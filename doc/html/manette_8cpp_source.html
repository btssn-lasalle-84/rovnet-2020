355  else if(maManetteDeplacement == virageAvantAGauche)
356  emit creationTrameDeplacement('A', puissanceGachetteDroite, 'G');
357  else if(maManetteDeplacement == virageAvantAGaucheDoucementCas1 || maManetteDeplacement == virageAvantAGaucheDoucementCas2)
358  emit creationTrameDeplacement('A', int(puissanceGachetteDroite * REDUCTION_VITESSE), 'G');
359  else if(maManetteDeplacement == virageAvantADroite)
360  emit creationTrameDeplacement('A', puissanceGachetteDroite, 'D');
361  else if(maManetteDeplacement == virageAvantADroiteDoucementCas1 || maManetteDeplacement == virageAvantADroiteDoucementCas2)
362  emit creationTrameDeplacement('A', int(puissanceGachetteDroite * REDUCTION_VITESSE), 'D');
363  else if(maManetteDeplacement == virageArriereAGauche)
